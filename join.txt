-- 1. Selezionare tutti gli studenti iscritti al Corso di Laurea in Economia

SELECT S.name, S.surname, s.registration_number, d.name
FROM students AS S
JOIN degrees AS D
WHERE D.name = "Corso di Laurea in Economia";

-- 2. Selezionare tutti i Corsi di Laurea Magistrale del Dipartimento di Neuroscienze

SELECT DP.name, DG.name, DG.level
FROM degrees AS DG
JOIN departments AS DP
WHERE DG.level = "magistrale"
AND DP.name = "Dipartimento di Neuroscienze";

or

SELECT DP.name, DG.name, DG.level
FROM degrees AS DG
JOIN departments AS DP ON DP.id = DG.department_id
WHERE DG.level = "magistrale"
AND DP.name = "Dipartimento di Neuroscienze";

-- 3. Selezionare tutti i corsi in cui insegna Fulvio Amato (id=44)

SELECT C.name, T.id, T.name, T.surname
FROM courses AS C
JOIN course_teacher AS CT ON C.id = CT.course_id
JOIN teachers AS T ON T.id = CT.teacher_id
WHERE T.id = 44;
